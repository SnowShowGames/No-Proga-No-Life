struct Azs
{
	char comp[4];
	int mark;
	int price;
};
int searchAzs(Azs b[], int n)
{
	int mn = b[0].price;
	for (int i = 0; i < n; i++)
	{
		if (mn < b[i].price) mn = b[i].price;
	}
	return mn;
}
